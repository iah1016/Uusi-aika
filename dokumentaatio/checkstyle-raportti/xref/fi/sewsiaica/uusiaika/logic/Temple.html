6   package fi.sewsiaica.uusiaika.logic;
7   
8   import fi.sewsiaica.uusiaika.domain.*;
9   import java.util.ArrayList;
10  import java.util.List;
11  import java.util.Map;
12  
13  /**
14   * The temple is where the magic happens. The winning conditions are checked
15   * here. Preaching to the congregation is also needed to keep the overall
16   * scepticism low.
17   *
18   * @author iah1016
19   */
20  public class Temple {
21  
22      private Map<String, Integer> intValues;
23      private int templeSceptDecr;
24      private int deathCultCharismaReq;
25      private int divineRightMoneyReq;
26  
27      /**
28       * The constructor set the values of templeSceptDecr, deathCultCharismaReq,
29       * and divineRightMoneyReq to its object variable.
30       *
31       * @param intValues Includes all the variable values of the game.
32       */
33      public Temple(Map<String, Integer> intValues) {
34          this.intValues = intValues;
35          this.templeSceptDecr = intValues.get("templeSceptDecr");
36          this.deathCultCharismaReq = intValues.get("templeDeathCultCharismaReq");
37          this.divineRightMoneyReq = intValues.get("templeDivineRightMoneyReq");
38      }
39  
40      /**
41       * As in every turn the members' scepticism increases, some impeding
42       * procedures ought to be carried out. Bring your A-game and deliver a
43       * lights-out-passionate performance to your precious flock.
44       *
45       * @param player You the man (or woman).
46       * @param sect Your greatest achievement.
47       * @return Currently, returns true unless the congregation size is zero.
48       */
49      public boolean preach(Player player, Sect sect) {
50          List<Villager> congregation = sect.getCongregation();
51          if (congregation == null) {
52              return false;
53          }
54  
55          if (congregation.size() > 0) {
56              for (int i = 0; i < congregation.size(); i++) {
57                  Villager member = congregation.get(i);
58                  int scept = member.getScepticism();
59                  member.setScepticism(scept - templeSceptDecr);
60              }
61              return true;
62          }
63          return false;
64      }
65  
66      /**
67       * This is it. No turning back. If your charisma is high enough, everyone is
68       * going to ascend to the next level.
69       *
70       * @param player The chief him- or herself.
71       * @return If you've high enough charisma, you win. If not, nothing happens.
72       */
73      public boolean offerSodaToAllMembers(Player player) {
74          return player.getCharisma() >= deathCultCharismaReq;
75      }
76  
77      /**
78       * A one-way ticket to Paradise (some obscure island in the western Pacific)
79       * and only you are going. You have taught your flock well and now they can
80       * manage themselves. You will take a reasonable reward of 100 percent of
81       * the Sect's balance with you.
82       *
83       * @param player The grand guru.
84       * @param sect Meh. They will do fine.
85       * @return If there is enough moolah in the bank, then you win. If not, then
86       * nothing happens.
87       */
88      public boolean buyTicketToParadiseIsland(Player player, Sect sect) {
89          return sect.getBalance() >= divineRightMoneyReq;
90      }
91  }

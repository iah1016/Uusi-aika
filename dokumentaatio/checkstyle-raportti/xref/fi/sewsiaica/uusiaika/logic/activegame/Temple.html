17  package fi.sewsiaica.uusiaika.logic.activegame;
18  
19  import fi.sewsiaica.uusiaika.domain.*;
20  import java.util.List;
21  import java.util.Map;
22  
23  /**
24   * The temple is where the magic happens. The winning conditions are checked
25   * here. Preaching to the congregation is also needed to keep the overall
26   * scepticism low.
27   *
28   * @author iah1016
29   */
30  public class Temple {
31  
32      private final Map<String, Integer> intValues;
33      private final int templeSceptDecr;
34      private final int deathCultCharismaReq;
35      private final int divineRightMoneyReq;
36  
37      /**
38       * The constructor set the values of templeSceptDecr, deathCultCharismaReq,
39       * and divineRightMoneyReq to its object variable.
40       *
41       * @param intValues Includes all the variable values of the game.
42       */
43      public Temple(Map<String, Integer> intValues) {
44          this.intValues = intValues;
45          this.templeSceptDecr = intValues.get("templeSceptDecr");
46          this.deathCultCharismaReq = intValues.get("templeDeathCultCharismaReq");
47          this.divineRightMoneyReq = intValues.get("templeDivineRightMoneyReq");
48      }
49  
50      /**
51       * As in every turn the members' scepticism increases, some impeding
52       * procedures ought to be carried out. Bring your A-game and deliver a
53       * lights-out-passionate performance to your precious flock.
54       *
55       * @param player You the man (or woman).
56       * @param sect Your greatest achievement.
57       * @return Currently, returns true unless the congregation size is zero.
58       */
59      public boolean preach(Player player, Sect sect) {
60          List<Villager> congregation = sect.getCongregation();
61          if (congregation == null) {
62              return false;
63          }
64  
65          if (congregation.size() > 0) {
66              for (int i = 0; i < congregation.size(); i++) {
67                  Villager member = congregation.get(i);
68                  int scept = member.getScepticism();
69                  member.setScepticism(scept - templeSceptDecr);
70              }
71              return true;
72          }
73          return false;
74      }
75  
76      /**
77       * This is it. No turning back. If your charisma is high enough, everyone is
78       * going to ascend to the next level.
79       *
80       * @param player The chief him- or herself.
81       * @return If you've high enough charisma, you win. If not, nothing happens.
82       */
83      public boolean offerSodaToAllMembers(Player player) {
84          return player.getCharisma() >= deathCultCharismaReq;
85      }
86  
87      /**
88       * A one-way ticket to Paradise (some obscure island in the western Pacific)
89       * and only you are going. You have taught your flock well and now they can
90       * manage themselves. You will take a reasonable reward of 100 percent of
91       * the Sect's balance with you.
92       *
93       * @param player The grand guru.
94       * @param sect Meh. They will do fine.
95       * @return If there is enough moolah in the bank, then you win. If not, then
96       * nothing happens.
97       */
98      public boolean buyTicketToParadiseIsland(Player player, Sect sect) {
99          return sect.getBalance() >= divineRightMoneyReq;
100     }
101 }

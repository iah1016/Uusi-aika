17  package fi.sewsiaica.uusiaika.logic.activegame;
18  
19  import fi.sewsiaica.uusiaika.logic.activegame.conversion.*;
20  import fi.sewsiaica.uusiaika.domain.*;
21  import java.util.ArrayList;
22  import java.util.List;
23  import java.util.Map;
24  import java.util.Random;
25  
26  /**
27   * ActiveGame is the game that is currently played.
28   *
29   * @author iah1016
30   */
31  public class ActiveGame {
32  
33      private final Map<String, Integer> configIntValues;
34      private final List<Villager> villagers;
35      private final Player player;
36      private final Sect sect;
37      private final ScoreHandler scoreHandler;
38      private Persuasion persuasion;
39      private Sermon sermon;
40      private Accusation accusation;
41      private TurnLogic turnLogic;
42      private Temple temple;
43      private TrainingCentre trainingCentre;
44      private List<Villager> targetVillagers;
45      private int gameEndingCondition;
46  
47      /**
48       * All the objects given as parameters will be permanently attached to this
49       * ActiveGame object.
50       *
51       * @param random Random is created by the Main class.
52       * @param configIntValues The variable values for all the logic modules
53       * attached to this ActiveGame.
54       * @param villagers The List of villagers.
55       * @param player The player of the active game.
56       * @param sect The player's sect.
57       */
58      public ActiveGame(Random random, Map<String, Integer> configIntValues,
59              List<Villager> villagers, Player player, Sect sect) {
60          this.configIntValues = configIntValues;
61          this.villagers = villagers;
62          this.player = player;
63          this.sect = sect;
64          this.scoreHandler = new ScoreHandler();
65          this.targetVillagers = new ArrayList<>();
66          this.gameEndingCondition = 0;
67          createLogicModules(random);
68          setMembersToCongregation();
69      }
70  
71      private void createLogicModules(Random random) {
72          this.turnLogic = new TurnLogic(configIntValues);
73          this.persuasion = new Persuasion(random, configIntValues,
74                  configIntValues.get("convMaxNumberOfPersuasions"),
75                  configIntValues.get("convPersPlayerBound"),
76                  configIntValues.get("convPersVilBound"));
77          this.sermon = new Sermon(random, configIntValues,
78                  configIntValues.get("convMaxNumberOfSermons"),
79                  configIntValues.get("convSermPlayerBound"),
80                  configIntValues.get("convSermVilBound"));
81          this.accusation = new Accusation(random, configIntValues,
82                  configIntValues.get("convMaxNumberOfAccusations"),
83                  configIntValues.get("convAccuPlayerBound"),
84                  configIntValues.get("convAccuVilBound"));
85          this.temple = new Temple(configIntValues);
86          this.trainingCentre = new TrainingCentre(configIntValues);
87      }
88  
89      private void setMembersToCongregation() {
90          for (Villager villager : villagers) {
91              if (villager.isInSect()) {
92                  sect.getCongregation().add(villager);
93              }
94          }
95      }

99          return turnLogic.getNumberOfTurns();

145 
146     @Override
147     public String toString() {
148         return "Turn: " + getNumberOfTurns() + "  " + player.getName()
149                 + "  " + sect.getName() + "  Balance: " + sect.getBalance()
150                 + "  Members: " + sect.getCongregation().size();
151     }
152 
153     /**
154      * The conditions are: [0] the game has not ended, [1] the maximum amount of
155      * turns has been reached, [2] the death cult ending, [3] the paradise
156      * island ending.
157      *
158      * @return Returns the ending condition.
159      */
160     public int getGameEndingCondition() {
161         return gameEndingCondition;
162     }
163 
164     public int getFinalScore() {
165         return scoreHandler.getScore();
166     }
167 
168     /**
169      * The active game ends with the given condition. The method calls the
170      * private method calculateEndScore(), which updates the end score with the
171      * aid of the ScoreHandler object.
172      *
173      * @param gameEndingCondition [1] the maximum amount of turns has been
174      * reached, [2] the death cult ending, [3] the paradise island ending.
175      * @return Returns the final score.
176      */
177     public int endThisActiveGame(int gameEndingCondition) {
178         this.gameEndingCondition = gameEndingCondition;
179         return calculateEndScore();
180     }
181 
182     private int calculateEndScore() {
183         int[] elements = new int[8];
184         elements[0] = this.gameEndingCondition;
185         elements[1] = player.getCharisma();
186         elements[2] = sect.getBalance();
187         elements[3] = sect.getCongregation().size();
188         elements[4] = getNumberOfTurns();
189         elements[5] = configIntValues.get("turnMaxNumberOfTurns");
190         elements[6] = configIntValues.get("templeDeathCultCharismaReq");
191         elements[7] = configIntValues.get("templeDivineRightMoneyReq");
192         return scoreHandler.calculateScore(elements);
193     }
194 
195     public Map<String, Integer> getConfigIntValues() {
196         return configIntValues;
197     }

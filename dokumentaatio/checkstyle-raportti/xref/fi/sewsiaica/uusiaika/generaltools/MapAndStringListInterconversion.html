17  package fi.sewsiaica.uusiaika.generaltools;
18  
19  import java.util.ArrayList;
20  import java.util.List;
21  import java.util.Map;
22  import java.util.Set;
23  
24  /**
25   * This abstract class contains the implementation for converting a
26   * String,?-type Map to a String-type List and most of the methods for the
27   * reverse conversion. The conversions from a List to a String,String-type or a
28   * String,Integer-type Map are implemented in the subclasses.
29   *
30   * @author iah1016
31   */
32  public abstract class MapAndStringListInterconversion {
33  
34      /**
35       * An abstract method for converting a String-type List to String,?-type
36       * Map, where ? is a wildcard.
37       *
38       * @param list A List of Strings.
39       * @return Returns a Map of an appropriate type, selected by the subclass.
40       */
41      public abstract Map<String, ?> convertStringListToMap(List<String> list);
42  
43      /**
44       * This method converts any String,?-type map to a String-type List, where ?
45       * is a wildcard.
46       *
47       * @param map Map's keys are of String type and values can be any type.
48       * @return A List of Strings is returned.
49       */
50      public List<String> convertMapToStringList(Map<String, ?> map) {
51          List<String> stringList = new ArrayList<>();
52          Set<String> keySet = map.keySet();
53  
54          for (String key : keySet) {
55              stringList.add(key + ": " + map.get(key).toString());
56          }
57  
58          return stringList;
59      }
60  
61      /**
62       * This method is used in convertStringListToMap. It processes one string
63       * from the StringList.
64       *
65       * @param string A string is given from the StringList.
66       * @return Returns a key and value pair for the Map.
67       */
68      public String[] processOneStringFromStringList(String string) {
69          boolean colonFound = false;
70          boolean postColonSpaceIgnored = false;
71          StringBuilder keySB = new StringBuilder();
72          StringBuilder valueSB = new StringBuilder();
73  
74          for (int charIdx = 0; charIdx < string.length(); charIdx++) {
75              char currentChar = string.charAt(charIdx);
76  
77              if (!colonFound) {
78                  if (!currentCharIsColon(currentChar)) {
79                      keySB.append(currentChar);
80                  } else {
81                      colonFound = true;
82                  }
83              } else if (!postColonSpaceIgnored) {
84                  postColonSpaceIgnored = true;
85              } else {
86                  valueSB.append(currentChar);
87              }
88              if (keySB.toString().isEmpty()) {
89                  return null;
90              }
91          }
92          String[] keyAndValue = {keySB.toString(),
93              returnValueOrNullIfValueIsEmpty(valueSB.toString())};
94          return keyAndValue;
95      }
96  
97      /**
98       * This method checks if the current char is a colon, which indicates that
99       * the key has been fully read.
100      *
101      * @param currentChar The character that is being checked from the string.
102      * @return True, if the character is a colon and false otherwise.
103      */
104     public boolean currentCharIsColon(char currentChar) {
105         return currentChar == ':';
106     }
107 
108     /**
109      * If there are no characters in the string after the colon (whitespace
110      * excluded), the value is null.
111      *
112      * @param value The value part of the string.
113      * @return Returns null if value is empty and value otherwise.
114      */
115     public String returnValueOrNullIfValueIsEmpty(String value) {
116         if (value.isEmpty()) {
117             return null;
118         }
119         return value;
120     }
121 }

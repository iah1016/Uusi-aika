19      public void openingView() {

23          playerAndSectNames[0] = reader.nextLine();

25          playerAndSectNames[1] = reader.nextLine();

27          game.initGame(playerAndSectNames);
28      }

30      public void mainMenu() {
31          while (true) {
32              System.out.println("Päävalikko:");
33              System.out.println("(a)  käännytysvalikkoon");
34              System.out.println("(x)  lopeta");
35              String option = reader.nextLine();
36              if (option.equals("x")) {
37                  break;
38              } else if (option.equals("a")) {
39                  conversionMenu(chooseVillagerMenu());
40              } else {
41                  System.out.println("Väärä komento!");
42              }
43          }
44      }
45  
46      public Villager chooseVillagerMenu() {
47          int index = 0;
48          while (true) {
49              System.out.println("valitse kyläläinen: 1-" + game.getVillagers().size());
50              String value = reader.nextLine();
51              if (value.matches("^[0-9]+$")) {
52                  index = Integer.parseInt(value) - 1;
53                  if (index >= 0 && index < game.getVillagers().size()) {
54                      System.out.println("kyläläinen " + game.getVillagers().get(index).getName() + " valittu.");
55                      return game.getVillagers().get(index);
56                  } else {
57                      System.out.println("virheellinen luku");
58                  }
59              } else {
60                  System.out.println("syötä mieluummin numero");
61              }
62  
63          }
64      }
65  
66      public void conversionMenu(Villager villager) {
67          boolean inTheLoop = true;
68  
69          while (inTheLoop) {
70              System.out.println("Käännytysvalikko:");
71              System.out.println("(a)  lepertele keveitä");
72              System.out.println("(b)  kerro pyhästä kalasta");
73              System.out.println("(c)  syytä syntisyydestä");
74              System.out.println("(m)  palaa päävalikkoon");
75              String option = reader.nextLine();
76              if (option.equals("m")) {
77                  System.out.println("ei sitten");
78                  inTheLoop = false;
79              } else if (option.equals("a") || 
80                      option.equals("b") ||
81                      option.equals("c")) {
82                  //a  charisma vs. selfAwareness + scepticism
83                  //b  charisma + argSkills vs. argSkills vs. scepticism
84                  //c  charisma vs. selfEsteem + argSkills
85                  inTheLoop = game.conversion(villager, option);
86              } else {
87                  System.out.println("Väärä komento!");
88                  inTheLoop = true;
89              }
90          }
91      }

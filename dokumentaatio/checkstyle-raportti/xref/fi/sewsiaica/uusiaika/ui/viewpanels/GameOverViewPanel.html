17  package fi.sewsiaica.uusiaika.ui.viewpanels;
18  
19  import fi.sewsiaica.uusiaika.logic.GameLogic;
20  import fi.sewsiaica.uusiaika.ui.GameFrame;
21  import fi.sewsiaica.uusiaika.ui.viewpanellisteners.GameOverViewPanelListener;
22  import java.awt.BorderLayout;
23  import java.awt.Color;
24  import java.awt.Dimension;
25  import java.awt.event.ActionListener;
26  import java.util.Map;
27  import javax.swing.AbstractButton;
28  import javax.swing.JPanel;
29  import javax.swing.JTextArea;
30  
31  /**
32   * This class extends AbstractViewPanel; its object displays the game over view.
33   *
34   * @author iah1016
35   */
36  public class GameOverViewPanel extends AbstractViewPanel {
37  
38      private final Dimension dimension;
39      private final GameLogic gameLogic;
40      private final GameFrame gameFrame;
41      private JPanel endMessagePanel;
42      private JPanel buttonPanel;
43      private final int endCondition;
44      private final int finalScore;
45  
46      /**
47       * Dimension, GameLogic, and GameFrame are given as parameters.
48       *
49       * @param dimension The dimensions of the panel.
50       * @param gameLogic The core logic of the game, through which the other
51       * logic parts are called.
52       * @param frame GameFrame gives itself as parameter, so that the active
53       * ViewPanel can be changed.
54       * @param endCondition The end conditions are: [0] the game has not ended,
55       * [1] the maximum amount of turns has been reached, [2] the death cult
56       * ending, [3] the paradise island ending.
57       * @param finalScore The final score.
58       */
59      public GameOverViewPanel(Dimension dimension, GameLogic gameLogic,
60              GameFrame frame, int endCondition, int finalScore) {
61          super();
62          this.dimension = dimension;
63          this.gameLogic = gameLogic;
64          this.gameFrame = frame;
65          this.endCondition = endCondition;
66          this.finalScore = finalScore;
67          this.updateComponents();
68      }
69  
70      @Override
71      public void updateComponents() {
72          Map<String, String> language = gameLogic.getActiveLanguage();
73          String[] textsForButtons = {
74              language.get("showHallOfFameButton"),
75              language.get("openingMenuViewButton")};
76  
77          endMessagePanel = createEndMessageTextPanePanel();
78          buttonPanel = super.getNewButtonPanel(textsForButtons);
79          this.addSubPanelsToViewPanel();
80      }
81  
82      @Override
83      protected final ActionListener createActionListener(
84              AbstractButton[] buttons) {
85          return new GameOverViewPanelListener(gameLogic, gameFrame, buttons);
86      }
87  
88      @Override
89      protected void addSubPanelsToViewPanel() {
90          this.setLayout(new BorderLayout());
91          this.add(endMessagePanel, BorderLayout.CENTER);
92          this.add(buttonPanel, BorderLayout.NORTH);
93      }
94  
95      private JPanel createEndMessageTextPanePanel() {
96          JPanel endMsgPanel = new JPanel();
97          JTextArea endMessageTextArea = new JTextArea();
98          endMessageTextAreaSettings(endMessageTextArea);
99  
100         endMsgPanel.setBackground(Color.decode("#a3c2c2"));
101         endMsgPanel.add(endMessageTextArea);
102         return endMsgPanel;
103     }
104     
105     private void endMessageTextAreaSettings(JTextArea endMessageTextArea) {
106         endMessageTextArea.setEditable(false);
107         endMessageTextArea.setText(pickEndMessage());
108         endMessageTextArea.setPreferredSize(new Dimension(600, 400));
109         endMessageTextArea.setWrapStyleWord(true);
110         endMessageTextArea.setLineWrap(true);
111         endMessageTextArea.setBackground(Color.decode("#ebebe0"));
112     }
113 
114     private String pickEndMessage() {
115         Map<String, String> language = gameLogic.getActiveLanguage();
116         String scoreString = language.get("finalScoreMessage") + finalScore
117                 + ".";
118         switch (endCondition) {
119             case 1:
120                 return language.get("endMessage1") + "\n" + scoreString;
121             case 2:
122                 return language.get("endMessage2") + "\n" + scoreString;
123             case 3:
124                 return language.get("endMessage3") + "\n" + scoreString;
125             default:
126                 return language.get("endMessageDefault") + "\n" + scoreString;
127         }
128     }
129 }

17  package fi.sewsiaica.uusiaika.ui.viewpanels;
18  
19  import java.awt.Color;
20  import java.awt.event.ActionListener;
21  import javax.swing.AbstractButton;
22  import javax.swing.BorderFactory;
23  import javax.swing.JButton;
24  import javax.swing.JPanel;
25  import javax.swing.JTextField;
26  
27  /**
28   * The abstract view panel currently contains the implementation of the
29   * createButtonPanel method.
30   *
31   * @author iah1016
32   */
33  public abstract class AbstractViewPanel extends JPanel {
34  
35      /**
36       * Calls the parent constructor with no arguments.
37       *
38       */
39      public AbstractViewPanel() {
40          super();
41      }
42  
43      /**
44       * This method returns a new Button subpanel (JPanel subclass) which
45       * contains AbstractButtons attached to the corresponding ActionListener. It
46       * uses the following methods: createButtons, createActionListener,
47       * createButtonPanel.
48       *
49       * @param textsForButtons The texts of the Buttons given as a string array.
50       * @return Returns the JPanel which contains the buttons.
51       */
52      protected JPanel getNewButtonPanel(String[] textsForButtons) {
53  
54          AbstractButton[] buttons = createButtons(textsForButtons);
55          ActionListener actionListener = createActionListener(buttons);
56          JPanel buttonPanel = createButtonPanel(
57                  buttons, actionListener);
58  
59          return buttonPanel;
60      }
61  
62      private JPanel createButtonPanel(AbstractButton[] buttons,
63              ActionListener actionListener) {
64          JPanel buttonPanel = new JPanel();
65  
66          for (AbstractButton button : buttons) {
67              button.addActionListener(actionListener);
68              buttonPanel.add(button);
69          }
70          applyButtonPanelSettings(buttonPanel);
71          return buttonPanel;
72      }
73  
74      private AbstractButton[] createButtons(String[] textsForButtons) {
75          AbstractButton[] buttons = new JButton[textsForButtons.length];
76  
77          for (int i = 0; i < buttons.length; i++) {
78              buttons[i] = new JButton(textsForButtons[i]);
79              applyButtonSettings(buttons[i]);
80          }
81          return buttons;
82      }
83  
84      private void applyButtonPanelSettings(JPanel buttonPanel) {
85          buttonPanel.setBackground(Color.decode("#a3c2c2"));
86      }
87  
88      private void applyButtonSettings(AbstractButton button) {
89          button.setBackground(Color.decode("#527a7a"));
90          button.setForeground(Color.WHITE);
91          button.setBorder(BorderFactory.createLineBorder(Color.decode("#527a7a"),
92                  7));
93      }
94  
95      /**
96       * The method creates a new JTextField and then adds it to the given
97       * textFieldPanel (JPanel subclass).
98       *
99       * @param textFieldPanel A JPanel subclass object.
100      * @param text The default text displayed in the text field.
101      * @param length The length of the text field.
102      * @return Returns a JTextField object.
103      */
104     protected JTextField createJTextField(JPanel textFieldPanel,
105             String text, int length) {
106         JTextField textField = new JTextField(length);
107         textField.setText(text);
108 
109         applyJTextFieldSettings(textField);
110         textFieldPanel.add(textField);
111         return textField;
112     }
113 
114     private void applyJTextFieldSettings(JTextField textField) {
115         textField.setBackground(Color.decode("#f0f0f5"));
116     }
117 
118     /**
119      * This method creates the ActionListener for the getNewButtonPanel method.
120      *
121      * @param buttons The AbstractButtons are given to the ActionListener's
122      * constructor as a parameter.
123      * @return Returns the ActionListener.
124      */
125     protected abstract ActionListener createActionListener(
126             AbstractButton[] buttons);
127 
128     /**
129      * This method updates the components.
130      */
131     public abstract void updateComponents();
132 
133     /**
134      * This method adds the sub panels to the view panel and makes them visible.
135      *
136      */
137     protected abstract void addSubPanelsToViewPanel();
138 }

17  package fi.sewsiaica.uusiaika.ui.viewpanellisteners;
18  
19  import fi.sewsiaica.uusiaika.logic.GameLogic;
20  import fi.sewsiaica.uusiaika.ui.GameFrame;
21  import fi.sewsiaica.uusiaika.ui.PanelNames;
22  import java.awt.event.ActionEvent;
23  import java.awt.event.ActionListener;
24  import javax.swing.AbstractButton;
25  
26  /**
27   * The object of this class handles the events in the temple view.
28   *
29   * @author iah1016
30   */
31  public class TempleViewPanelListener implements ActionListener {
32  
33      private final GameFrame gameFrame;
34      private final GameLogic gameLogic;
35      private final AbstractButton preachButton;
36      private final AbstractButton offerSodaButton;
37      private final AbstractButton buyTicketButton;
38      private final AbstractButton returnToMapViewButton;
39      private final AbstractButton endTurnButton;
40  
41      /**
42       * The constructor is given an array of five AbstractButtons, GameFrame, and
43       * GameLogic as parameters.
44       *
45       * @param gameLogic The core logic of the game, through which the other
46       * logic parts are called.
47       * @param frame The core class of the GUI. It controls which view panel is
48       * shown.
49       * @param buttons Buttons 0-2 call GameLogic's templeActions method with the
50       * option to preach, offer soda to everyone, or buy a one-way ticket
51       * respectively. The latter two will end the game if the conditions are met.
52       * [3] changes the active view to the map view and [4] ends the turn.
53       */
54      public TempleViewPanelListener(GameLogic gameLogic, GameFrame frame,
55              AbstractButton[] buttons) {
56          this.gameFrame = frame;
57          this.gameLogic = gameLogic;
58          this.preachButton = buttons[0];
59          this.offerSodaButton = buttons[1];
60          this.buyTicketButton = buttons[2];
61          this.returnToMapViewButton = buttons[3];
62          this.endTurnButton = buttons[4];
63      }
64  
65      @Override
66      public void actionPerformed(ActionEvent ae) {
67  
68          if (ae.getSource() == returnToMapViewButton) {
69              gameFrame.changeViewPanel(PanelNames.MAP_VIEW);
70          } else if (ae.getSource() == preachButton) {
71              preachSelected();
72          } else if (ae.getSource() == offerSodaButton) {
73              offerSodaSelected();
74          } else if (ae.getSource() == buyTicketButton) {
75              buyTicketSelected();
76          } else if (ae.getSource() == endTurnButton) {
77              if (gameLogic.endTurn()) {
78                  updateView();
79              } else {
80                  gameLogic.getActiveGame().setGameEndingCondition(1);
81                  gameFrame.changeViewPanel(PanelNames.GAME_OVER_VIEW);
82              }
83          }
84      }
85  
86      private void updateView() {
87          gameFrame.changeViewPanel(PanelNames.TEMPLE_VIEW);
88      }
89  
90      private void preachSelected() {
91          if (gameLogic.templeActions('a')) {
92              System.out.println("Praise the Holy Fish! "
93                      + "Your performance was a success. "
94                      + "Everyone is a little less sceptical now.");
95          } else {
96              System.out.println("A solid perfomance. "
97                      + "Too bad there was no one here to hear it!");
98          }
99          updateView();
100     }
101 
102     private void offerSodaSelected() {
103         if (gameLogic.templeActions('b')) {
104             System.out.println("With you guidance, your flock will "
105                     + "take the daring step to ascend to the next level.");
106             gameLogic.getActiveGame().setGameEndingCondition(2);
107             gameFrame.changeViewPanel(PanelNames.GAME_OVER_VIEW);
108         } else {
109             System.out.println("You lack charisma to pull this off.");
110             updateView();
111         }
112     }
113 
114     private void buyTicketSelected() {
115         if (gameLogic.templeActions('c')) {
116             System.out.println("A one-way ticket to Paradise (some obscure "
117                     + "island in the western Pacific) and only you are "
118                     + "going.\n"
119                     + "You have taught your flock well and now they can "
120                     + "manage themselves.\nYou will take a reasonable "
121                     + "reward of 100 percent of the Sect's balance with "
122                     + "you.");
123             gameLogic.getActiveGame().setGameEndingCondition(3);
124             gameFrame.changeViewPanel(PanelNames.GAME_OVER_VIEW);
125         } else {
126             System.out.println("There's not enough money on the account.");
127             updateView();
128         }
129     }
130 }

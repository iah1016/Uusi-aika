17  package fi.sewsiaica.uusiaika.ui.subpanels;
18  
19  import fi.sewsiaica.uusiaika.domain.Villager;
20  import java.awt.Color;
21  import java.awt.Component;
22  import java.util.List;
23  import javax.swing.DefaultListCellRenderer;
24  import javax.swing.JList;
25  
26  /**
27   * This class extends DefaultListCellRenderer with a minor modification: the
28   * sect members are coloured in magenta in the villager list panel.
29   *
30   * @author iah1016
31   */
32  public class CustomCellRendererForVillagerListPanel
33          extends DefaultListCellRenderer {
34  
35      private final List<Villager> allVillagers;
36  
37      /**
38       * The constructor is given the list of all villagers which it sets as its
39       * object variable.
40       *
41       * @param allVillagers The list of all villagers.
42       */
43      public CustomCellRendererForVillagerListPanel(List<Villager> allVillagers) {
44          this.allVillagers = allVillagers;
45      }
46  
47      @Override
48      public Component getListCellRendererComponent(
49              JList<?> jlist, Object o, int idx, boolean bln, boolean bln1) {
50          Component component
51                  = super.getListCellRendererComponent(jlist, o, idx, bln, bln1);
52  
53          boolean[] inSectArray = updateinSectArray();
54          if (inSectArray[idx]) {
55              component.setForeground(Color.decode("#993366"));
56          }
57          return component;
58      }
59  
60      private boolean[] updateinSectArray() {
61          boolean[] inSectArray = new boolean[allVillagers.size()];
62          for (int i = 0; i < inSectArray.length; i++) {
63              inSectArray[i] = allVillagers.get(i).isInSect();
64          }
65          return inSectArray;
66      }
67  }

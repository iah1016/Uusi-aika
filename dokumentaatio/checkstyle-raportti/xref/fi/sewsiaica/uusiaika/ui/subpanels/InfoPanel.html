17  package fi.sewsiaica.uusiaika.ui.subpanels;
18  
19  import fi.sewsiaica.uusiaika.domain.Villager;
20  import fi.sewsiaica.uusiaika.logic.GameLogic;
21  import fi.sewsiaica.uusiaika.logic.activegame.ActiveGame;
22  import java.awt.Color;
23  import java.util.List;
24  import javax.swing.JTextPane;
25  
26  /**
27   * The info panel shows the following data: Turn number, Player name, Sect name,
28   * Sect balance, and the number of members in the sect.
29   *
30   * @author iah1016
31   */
32  public class InfoPanel extends AbstractSubPanel {
33  
34      private final GameLogic gameLogic;
35  
36      /**
37       * The class gets all the information from ActiveGame via GameLogic. The
38       * latter is given as a parameter to the constructor.
39       *
40       * @param gameLogic The core logic of the game, through which the other
41       * logic parts are called.
42       */
43      public InfoPanel(GameLogic gameLogic) {
44          super(gameLogic);
45          this.gameLogic = gameLogic;
46          addContentOnlyIfActiveGameIsNotNull();
47      }
48  
49      @Override
50      protected void addContents() {
51          JTextPane text = new JTextPane();
52          text.setText(buildShownString());
53          add(text);
54          
55          text.setForeground(Color.WHITE);
56          text.setBackground(Color.decode("#52527a"));
57          setBackground(Color.decode("#52527a"));
58      }
59  
60      private String buildShownString() {
61          ActiveGame activeGame = gameLogic.getActiveGame();
62          List<Villager> targetVillagers = activeGame.getTargetVillagers();
63  
64          StringBuilder stringBuilder = new StringBuilder();
65          stringBuilder.append(activeGame.toString());
66  
67          if (!targetVillagers.isEmpty()) {
68              int totalTargets = targetVillagers.size();
69              String nextTarget = targetVillagers.get(0).toString();
70              stringBuilder.append("  current target: ").append(nextTarget)
71                      .append("  total targets: ").append(totalTargets);
72          }
73  
74          return stringBuilder.toString();
75      }
76  
77  }

6   package fi.sewsiaica.uusiaika.io;

8   import java.io.File;
9   import java.io.FileNotFoundException;
10  import java.util.ArrayList;
11  import java.util.HashMap;
12  import java.util.List;
13  import java.util.Map;
14  import java.util.Scanner;
15  
16  /**
17   * This is a basic file reader that reads every line from a text file but
18   * discards the empty lines and the lines that contain a space or a hash sign as
19   * the first character. All the other lines are returned in an ArrayList of
20   * Strings.
21   *
22   * @author iah1016
23   */
24  public class ReadFromTextFile {
25  
26      private Scanner fileReader;
27  
28      /**
29       * Scanner object is brought from the outside for easier testing.
30       *
31       * @param reader
32       */
33      public ReadFromTextFile(Scanner reader) {
34          this.fileReader = reader;
35      }
36  
37      /**
38       * The method yanks lines from a text file. The yankees are transferred out
39       * in an ArrayList. Hashed or spaced out (comment) lines are excluded.
40       *
41       * @param textfile
42       * @return
43       * @throws FileNotFoundException
44       */
45      public List<String> yankTextFromFile(File textfile) throws
46              FileNotFoundException {
47          fileReader = new Scanner(textfile);
48          List<String> newLines = new ArrayList<>();
49  
50          while (fileReader.hasNextLine()) {
51              String line = fileReader.nextLine();
52              if (!isIgnorableLine(line)) {
53                  newLines.add(line);
54              }
55          }
56          fileReader.close();
57          return returnNewLineIffReaderIsClosed(newLines);
58      }
59  
60      /**
61       * Checking if the line is empty or has a space or a hash sign in the
62       * beginning.
63       *
64       * @param line A line that has been yanked from a text file.
65       * @return If one of the mentioned conditions is met, the line is ignored.
66       */
67      public boolean isIgnorableLine(String line) {
68          if (line.isEmpty()) {
69              return true;
70          }
71          return line.charAt(0) == ' ' || line.charAt(0) == '#';
72      }
73  
74      /**
75       * A fairly superfluous method for an earnest antimutator. Its purpose is to
76       * confirm that the scanner has been closed.
77       *
78       * @param list All them beautiful yankees.
79       * @return If everything goes as expected (ie. you are not an evil pit
80       * mutant), the method returns the list. If you are an evil pit mutant, null
81       * is all you get here.
82       */
83      public List<String> returnNewLineIffReaderIsClosed(
84              List<String> list) {
85          try {
86              fileReader.hasNext();
87          } catch (Exception e) {
88              return list;
89          }
90          return null;
91      }
92  }

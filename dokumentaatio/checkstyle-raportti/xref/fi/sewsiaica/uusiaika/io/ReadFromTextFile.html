17  package fi.sewsiaica.uusiaika.io;
18  
19  import java.io.File;
20  import java.io.FileNotFoundException;
21  import java.util.ArrayList;
22  import java.util.List;
23  import java.util.Scanner;
24  
25  /**
26   * This is a basic file reader that reads every line from a text file but
27   * discards the empty lines and the lines that contain a space or a hash sign as
28   * the first character. All the other lines are returned in an ArrayList of
29   * Strings.
30   *
31   * @author iah1016
32   */
33  public class ReadFromTextFile {
34  
35      /**
36       * The method yanks lines from a text file. The yankees are transferred out
37       * in a String-List. Hashed or spaced out (comment) lines are excluded.
38       *
39       * @param textfile The yanking target.
40       * @return The yanked lines of goodness.
41       * @throws FileNotFoundException Throws the FileNotFoundException.
42       */
43      public List<String> yankTextFromFile(File textfile) throws
44              FileNotFoundException {
45          Scanner fileReader = new Scanner(textfile, "UTF-8");
46          List<String> newLines = new ArrayList<>();
47  
48          while (fileReader.hasNextLine()) {
49              String line = fileReader.nextLine();
50              if (!isIgnorableLine(line)) {
51                  newLines.add(line);
52              }
53          }
54          fileReader.close();
55          return returnNewLineIffReaderIsClosed(newLines, fileReader);
56      }
57  
58      /**
59       * Checking if the line is empty or has a space or a hash sign in the
60       * beginning.
61       *
62       * @param line A line that has been yanked from a text file.
63       * @return If one of the mentioned conditions is met, the line is ignored.
64       */
65      public boolean isIgnorableLine(String line) {
66          if (line.isEmpty()) {
67              return true;
68          }
69          return line.charAt(0) == ' ' || line.charAt(0) == '#';
70      }
71  
72      /**
73       * A fairly superfluous method for an earnest antimutator. Its purpose is to
74       * confirm that the scanner has been closed.
75       *
76       * @param list All them beautiful yankees.
77       * @param reader Scanner that should closed already.
78       * @return If everything goes as expected (ie. you are not an evil pit
79       * mutant), the method returns the list. If you are an evil pit mutant, null
80       * is all you get here.
81       */
82      public List<String> returnNewLineIffReaderIsClosed(
83              List<String> list, Scanner reader) {
84          try {
85              reader.hasNext();
86          } catch (Exception e) {
87              return list;
88          }
89          return null;
90      }
91  }
